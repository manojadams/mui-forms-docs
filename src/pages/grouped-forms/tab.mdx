## Usage

```tsx
import React from "react";
import schema from "./schema.json";
import MuiForms from "mui-forms";

function MyTab() {
    return (
        <MuiForms
            schema={schema}
            onSubmit={() => {
                // do nothing
            }}
        />
    );
}
```
